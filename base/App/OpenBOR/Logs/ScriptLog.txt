
#### animationscript function main #####
# data/chars/hien/hien.txt
########################################
void main()
{
    int frame = getlocalvar("frame");
    int animhandle = getlocalvar("animhandle");
    if(animhandle==16)
    {

if(frame == 1){
void vSelf = getlocalvar("self");
int   iMP = getentityproperty(vSelf, "MP");
changeentityproperty(vSelf, "MP", iMP - 10);
}
        return;
    }
    if(animhandle==19)
    {

if(frame == 1){
void vSelf = getlocalvar("self");
int   iMP = getentityproperty(vSelf, "MP");
changeentityproperty(vSelf, "MP", iMP - 10);
}
        return;
    }

}

#### animationscript function main #####
# data/chars/enyest/enyest.txt
########################################
void main()
{
    int frame = getlocalvar("frame");
    int animhandle = getlocalvar("animhandle");
    if(animhandle==47)
    {

    void self = getlocalvar("self");
    void MP = getentityproperty(self,"mp");
    if (frame == 0){   
      if (MP < 10){
        changeentityproperty(self, "animation", openborconstant("ANI_FOLLOW1"));
        changeentityproperty(self, "animpos", 0);
      }
    }
    if (frame == 1){
      changeentityproperty(self, "mp", MP-10);
    }
        return;
    }
    if(animhandle==50)
    {

    void self = getlocalvar("self");
    void MP = getentityproperty(self,"mp");
    if (frame == 0){   
      if (MP < 10){
        changeentityproperty(self, "animation", openborconstant("ANI_FOLLOW1"));
        changeentityproperty(self, "animpos", 0);
      }
    }
    if (frame == 1){
      changeentityproperty(self, "mp", MP-40);
    }
        return;
    }

}
