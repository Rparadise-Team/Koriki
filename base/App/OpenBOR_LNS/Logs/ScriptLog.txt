void main()
{
    int frame = getlocalvar("frame");
    int animhandle = getlocalvar("animhandle");

    {
	
		void vSelf = getlocalvar("self");	
		 if(frame==0){

				 changeentityproperty(vSelf, "animation", openborconstant("ANI_FOLLOW1"));
				
			}
    }
        return;
    }
    if(animhandle==3)
    {
	
				int id;
				
		 if(frame==0){
      if(getsaveinfo(0, "times_completed") + getsaveinfo(1, "times_completed") + getsaveinfo(2, "times_completed") + getsaveinfo(3, "times_completed") + getsaveinfo(4, "times_completed")  >= 13){
        id = loadsample("data/sounds/evil-laugh.wav", 1);
				playsample(id, 0, 120, 120, 100, 0);
				
			}
    }
				
    if(frame==18){
      if(getsaveinfo(0, "times_completed") + getsaveinfo(1, "times_completed") + getsaveinfo(2, "times_completed") + getsaveinfo(3, "times_completed") + getsaveinfo(4, "times_completed") >= 13){
      	unloadsample(id);
      	setglobalvar("difficulty", 5);
        jumptobranch("GOARMMA",1);
      }
    }
        return;
    }

}

      int  iEntity;                                       //Entity enumeration holder.
      char iName;                                         //Entity Name.
      int  iMax      = openborvariant("ent_max");         //Entity count.

      //Enumerate and loop through entity collection.
      for(iEntity=0; iEntity<iMax; iEntity++){
        vEntity = getentity(iEntity);                 //Get target entity from current loop.

          killentity(vEntity);
        }
      }
    }
    if(frame==2){
      killentity(self);
    }
        return;
    }
